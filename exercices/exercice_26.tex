  \ln(56)-\ln(7)+\ln(4) &=& \ln(7\times2^3)-\ln(7)+\ln(2^2) \\
  &=& \ln(7) + 3\ln(2)-\ln(7)+2\ln(2) \\

b) \begin{eqnarray*}
  \ln(\sqrt{216}) &=& \ln(\sqrt{6^3}) \\
  &=& \ln(6^\frac{3}{2}) \\
  &=& \frac{3}{2}\ln(6)
\end{eqnarray*}
